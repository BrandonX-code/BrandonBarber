# Usa imagen base de .NET 9 (puedes cambiarla si estás usando preview)
FROM mcr.microsoft.com/dotnet/aspnet:9.0-preview AS base
WORKDIR /app
EXPOSE 80

FROM mcr.microsoft.com/dotnet/sdk:9.0-preview AS build
WORKDIR /src
COPY ["BrandonBarber.csproj", "./"]
RUN dotnet restore "./BrandonBarber.csproj"
COPY . .
RUN dotnet build "./BrandonBarber.csproj" -c Release -o /app/build

FROM build AS publish
RUN dotnet publish "./BrandonBarber.csproj" -c Release -o /app/publish

FROM base AS final
WORKDIR /app
COPY --from=publish /app/publish .
ENTRYPOINT ["dotnet", "BrandonBarber.dll"]
